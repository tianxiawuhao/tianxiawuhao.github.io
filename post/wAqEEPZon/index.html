еңЁжң¬з« дёӯпјҢжҲ‘д»¬е°ҶеӯҰд№ еҰӮдҪ•е°ҶKafkaдёҺApache StormйӣҶжҲҗгҖӮ

е…ідәҺStorm

StormжңҖеҲқз”ұNathan Marzе’ҢBackTypeзҡ„еӣўйҳҹеҲӣе»әгҖӮ еңЁзҹӯж—¶й—ҙеҶ…пјҢApache StormжҲҗдёәеҲҶеёғејҸе®һж—¶еӨ„зҗҶзі»з»ҹзҡ„ж ҮеҮҶпјҢе…Ғи®ёжӮЁеӨ„зҗҶеӨ§йҮҸж•°жҚ®гҖӮ StormжҳҜйқһеёёеҝ«зҡ„пјҢе№¶дё”дёҖдёӘеҹәеҮҶж—¶й’ҹдёәжҜҸдёӘиҠӮзӮ№жҜҸз§’еӨ„зҗҶи¶…иҝҮдёҖзҷҫдёҮдёӘе…ғз»„гҖӮ Apache StormжҢҒз»ӯиҝҗиЎҢпјҢд»Һй…ҚзҪ®зҡ„жәҗ(Spouts)ж¶ҲиҖ—ж•°жҚ®пјҢе№¶е°Ҷж•°жҚ®дј йҖ’еҲ°еӨ„зҗҶз®ЎйҒ“(Bolts)гҖӮ иҒ”еҗҲпјҢSpoutsе’ҢBoltжһ„жҲҗдёҖдёӘжӢ“жү‘гҖӮ

дёҺStormйӣҶжҲҗ

Kafkaе’ҢStormиҮӘз„¶дә’иЎҘпјҢе®ғд»¬ејәеӨ§зҡ„еҗҲдҪңиғҪеӨҹе®һзҺ°еҝ«йҖҹз§»еҠЁзҡ„еӨ§ж•°жҚ®зҡ„е®һж—¶жөҒеҲҶжһҗгҖӮ Kafkaе’ҢStormйӣҶжҲҗжҳҜдёәдәҶдҪҝејҖеҸ‘дәәе‘ҳжӣҙе®№жҳ“ең°д»ҺStormжӢ“жү‘иҺ·еҸ–е’ҢеҸ‘еёғж•°жҚ®жөҒгҖӮ

жҰӮеҝөжөҒ

SpoutsжҳҜжөҒзҡ„жәҗгҖӮ дҫӢеҰӮпјҢдёҖдёӘе–·еӨҙеҸҜд»Ҙд»ҺKafka TopicиҜ»еҸ–е…ғз»„е№¶е°Ҷе®ғд»¬дҪңдёәжөҒеҸ‘йҖҒгҖӮ Boltж¶ҲиҖ—иҫ“е…ҘжөҒпјҢеӨ„зҗҶе№¶еҸҜиғҪеҸ‘е°„ж–°зҡ„жөҒгҖӮ BoltеҸҜд»Ҙд»ҺиҝҗиЎҢеҮҪж•°пјҢиҝҮж»Өе…ғз»„пјҢжү§иЎҢжөҒиҒҡеҗҲпјҢжөҒиҝһжҺҘпјҢдёҺж•°жҚ®еә“дәӨи°ҲзӯүзӯүеҒҡд»»дҪ•дәӢжғ…гҖӮ StormжӢ“жү‘дёӯзҡ„жҜҸдёӘиҠӮзӮ№е№¶иЎҢжү§иЎҢгҖӮ жӢ“жү‘ж— йҷҗиҝҗиЎҢпјҢзӣҙеҲ°з»Ҳжӯўе®ғгҖӮ Stormе°ҶиҮӘеҠЁйҮҚж–°еҲҶй…Қд»»дҪ•еӨұиҙҘзҡ„д»»еҠЎгҖӮ жӯӨеӨ–пјҢStormдҝқиҜҒжІЎжңүж•°жҚ®дёўеӨұпјҢеҚідҪҝжңәеҷЁеҒңжңәе’Ңж¶ҲжҒҜиў«дёўејғгҖӮ

и®©жҲ‘д»¬иҜҰз»ҶдәҶи§ЈKafka-StormйӣҶжҲҗAPIгҖӮ жңүдёүдёӘдё»иҰҒзұ»йӣҶжҲҗKafkaдёҺStormгҖӮ д»–д»¬еҰӮдёӢ -

BrokerHosts - ZkHosts & StaticHosts

BrokerHostsжҳҜдёҖдёӘжҺҘеҸЈпјҢZkHostsе’ҢStaticHostsжҳҜе®ғзҡ„дёӨдёӘдё»иҰҒе®һзҺ°гҖӮ ZkHostsз”ЁдәҺйҖҡиҝҮеңЁZooKeeperдёӯз»ҙжҠӨз»ҶиҠӮжқҘеҠЁжҖҒи·ҹиёӘKafkaд»ЈзҗҶпјҢиҖҢStaticHostsз”ЁдәҺжүӢеҠЁ/йқҷжҖҒи®ҫзҪ®Kafkaд»ЈзҗҶеҸҠе…¶иҜҰз»ҶдҝЎжҒҜгҖӮ ZkHostsжҳҜи®ҝй—®Kafkaд»ЈзҗҶзҡ„з®ҖеҚ•еҝ«жҚ·зҡ„ж–№ејҸгҖӮ

SchemeAsMultiSchemeжҳҜдёҖдёӘжҺҘеҸЈпјҢз”ЁдәҺжҢҮзӨәеҰӮдҪ•е°Ҷд»ҺKafkaдёӯж¶ҲиҖ—зҡ„ByteBufferиҪ¬жҚўдёәйЈҺжҡҙе…ғз»„гҖӮ е®ғжәҗиҮӘMultiSchemeе№¶жҺҘеҸ—Schemeзұ»зҡ„е®һзҺ°гҖӮ жңүеҫҲеӨҡSchemeзұ»зҡ„е®һзҺ°пјҢдёҖдёӘиҝҷж ·зҡ„е®һзҺ°жҳҜStringSchemeпјҢе®ғе°Ҷеӯ—иҠӮи§ЈжһҗдёәдёҖдёӘз®ҖеҚ•зҡ„еӯ—з¬ҰдёІгҖӮ е®ғиҝҳжҺ§еҲ¶иҫ“еҮәеӯ—ж®өзҡ„е‘ҪеҗҚгҖӮ зӯҫеҗҚе®ҡд№үеҰӮдёӢгҖӮ

еҲӣе»әBolt

BoltжҳҜдёҖдёӘдҪҝз”Ёе…ғз»„дҪңдёәиҫ“е…ҘпјҢеӨ„зҗҶе…ғз»„пјҢе№¶дә§з”ҹж–°зҡ„е…ғз»„дҪңдёәиҫ“еҮәзҡ„з»„д»¶гҖӮ Boltе°Ҷе®һзҺ°IRichBoltжҺҘеҸЈгҖӮ еңЁжӯӨзЁӢеәҸдёӯпјҢдҪҝз”ЁдёӨдёӘBoltзұ»WordSplitter-Boltе’ҢWordCounterBoltжқҘжү§иЎҢж“ҚдҪңгҖӮ

IRichBoltжҺҘеҸЈжңүд»ҘдёӢж–№жі• -

  • еҮҶеӨҮ - дёәBoltжҸҗдҫӣиҰҒжү§иЎҢзҡ„зҺҜеўғгҖӮ жү§иЎҢеҷЁе°ҶиҝҗиЎҢжӯӨж–№жі•жқҘеҲқе§ӢеҢ–е–·еӨҙгҖӮ
  • жү§иЎҢ - еӨ„зҗҶеҚ•дёӘе…ғз»„зҡ„иҫ“е…ҘгҖӮ
  • жё…зҗҶ - еҪ“BoltиҰҒе…ій—ӯж—¶и°ғз”ЁгҖӮ
  • declareOutputFields - еЈ°жҳҺе…ғз»„зҡ„иҫ“еҮәжЁЎејҸгҖӮ

и®©жҲ‘д»¬еҲӣе»әSplitBolt.javaпјҢе®ғе®һзҺ°йҖ»иҫ‘еҲҶеүІдёҖдёӘеҸҘеӯҗеҲ°иҜҚе’ҢCountBolt.javaпјҢе®ғе®һзҺ°йҖ»иҫ‘еҲҶзҰ»зӢ¬зү№зҡ„еҚ•иҜҚе’Ңи®Ўж•°е…¶еҮәзҺ°гҖӮ

SplitBolt.java

import java.util.Map;

import backtype.storm.tuple.Tuple;
import backtype.storm.tuple.Fields;
import backtype.storm.tuple.Values;

import backtype.storm.task.OutputCollector;
import backtype.storm.topology.OutputFieldsDeclarer;
import backtype.storm.topology.IRichBolt;
import backtype.storm.task.TopologyContext;

public class SplitBolt implements IRichBolt {
   private OutputCollector collector;
   
   @Override
   public void prepare(Map stormConf, TopologyContext context,
      OutputCollector collector) {
      this.collector = collector;
   }
   
   @Override
   public void execute(Tuple input) {
      String sentence = input.getString(0);
      String[] words = sentence.split(" ");
      
      for(String word: words) {
         word = word.trim();
         
         if(!word.isEmpty()) {
            word = word.toLowerCase();
            collector.emit(new Values(word));
         }
         
      }

      collector.ack(input);
   }
   
   @Override
   public void declareOutputFields(OutputFieldsDeclarer declarer) {
      declarer.declare(new Fields("word"));
   }

   @Override
   public void cleanup() {}
   
   @Override
   public Map<String, Object> getComponentConfiguration() {
      return null;
   }
   
}

CountBolt.java

import java.util.Map;
import java.util.HashMap;

import backtype.storm.tuple.Tuple;
import backtype.storm.task.OutputCollector;
import backtype.storm.topology.OutputFieldsDeclarer;
import backtype.storm.topology.IRichBolt;
import backtype.storm.task.TopologyContext;

public class CountBolt implements IRichBolt{
   Map<String, Integer> counters;
   private OutputCollector collector;
   
   @Override
   public void prepare(Map stormConf, TopologyContext context,
   OutputCollector collector) {
      this.counters = new HashMap<String, Integer>();
      this.collector = collector;
   }

   @Override
   public void execute(Tuple input) {
      String str = input.getString(0);
      
      if(!counters.containsKey(str)){
         counters.put(str, 1);
      }else {
         Integer c = counters.get(str) +1;
         counters.put(str, c);
      }
   
      collector.ack(input);
   }

   @Override
   public void cleanup() {
      for(Map.Entry<String, Integer> entry:counters.entrySet()){
         System.out.println(entry.getKey()&plus;" : " &plus; entry.getValue());
      }
   }

   @Override
   public void declareOutputFields(OutputFieldsDeclarer declarer) {
   
   }

   @Override
   public Map<String, Object> getComponentConfiguration() {
      return null;
   }
}

KafkaStormSample.java

import backtype.storm.Config;
import backtype.storm.LocalCluster;
import backtype.storm.topology.TopologyBuilder;

import java.util.ArrayList;
import java.util.List;
import java.util.UUID;

import backtype.storm.spout.SchemeAsMultiScheme;
import storm.kafka.trident.GlobalPartitionInformation;
import storm.kafka.ZkHosts;
import storm.kafka.Broker;
import storm.kafka.StaticHosts;
import storm.kafka.BrokerHosts;
import storm.kafka.SpoutConfig;
import storm.kafka.KafkaConfig;
import storm.kafka.KafkaSpout;
import storm.kafka.StringScheme;

public class KafkaStormSample {
   public static void main(String[] args) throws Exception{
      Config config = new Config();
      config.setDebug(true);
      config.put(Config.TOPOLOGY_MAX_SPOUT_PENDING, 1);
      String zkConnString = "localhost:2181";
      String topic = "my-first-topic";
      BrokerHosts hosts = new ZkHosts(zkConnString);
      
      SpoutConfig kafkaSpoutConfig = new SpoutConfig (hosts, topic, "/" + topic,    
         UUID.randomUUID().toString());
      kafkaSpoutConfig.bufferSizeBytes = 1024 * 1024 * 4;
      kafkaSpoutConfig.fetchSizeBytes = 1024 * 1024 * 4;
      kafkaSpoutConfig.forceFromStart = true;
      kafkaSpoutConfig.scheme = new SchemeAsMultiScheme(new StringScheme());

      TopologyBuilder builder = new TopologyBuilder();
      builder.setSpout("kafka-spout", new KafkaSpout(kafkaSpoutCon-fig));
      builder.setBolt("word-spitter", new SplitBolt()).shuffleGroup-ing("kafka-spout");
      builder.setBolt("word-counter", new CountBolt()).shuffleGroup-ing("word-spitter");
         
      LocalCluster cluster = new LocalCluster();
      cluster.submitTopology("KafkaStormSample", config, builder.create-Topology());

      Thread.sleep(10000);
      
      cluster.shutdown();
   }
}
